
import React, { useState, useRef, useEffect } from 'react';
import { 
  Upload, Image as ImageIcon, Wand2, Sparkles, Download, ArrowRight, Layout, AlertCircle, 
  RefreshCw, PenTool, User as UserIcon, Crown, LogOut, MessageSquare, Target, Package, 
  ShieldCheck, Square, Sun, Gem, Camera, Leaf, Zap, History, Home, Droplets, Palette, Coffee, Cpu,
  Info, Beaker, Globe, Type, List, Trash2, XCircle, UserRound, Shirt, MapPin, Plus, X, Layers,
  AlignLeft, AlignCenter, AlignRight, Coins, Star, LayoutGrid, CheckCircle2, Lightbulb, AlertTriangle
} from 'lucide-react';
import { Button } from './components/Button';
import { ComparisonSlider } from './components/ComparisonSlider';
import { AuthModal } from './components/AuthModal';

import { UserProfileModal } from './components/UserProfileModal';
import { AdminDashboard } from './components/AdminDashboard';
import { Footer } from './components/Footer';
import { STYLES, INFOGRAPHIC_STYLES, TEMPLATES, MARKETPLACE_PACKAGE_COST, FOTOSESIYA_PRO_COST_PER_IMAGE, INFOGRAPHIC_AI_IDEAS, TOKEN_COST_PER_GENERATION, TOKEN_COST_PER_REGENERATION } from './constants';
import { AppMode, VisualStyle, MarketplaceTemplate, ImageSize, User, HistoryItem, InfographicDensity } from './types';
import * as gemini from './services/gemini';
import { UserService } from './services/db';
import { useLanguage } from './LanguageContext';
import SmartMatn from './components/SmartMatn';

import { AdminOps } from './components/AdminOps';

const IconMap: Record<string, React.ReactNode> = {
  Square: <Square size={14} />,
  Sun: <Sun size={14} />,
  Gem: <Gem size={14} />,
  Camera: <Camera size={14} />,
  Leaf: <Leaf size={14} />,
  Zap: <Zap size={14} />,
  History: <History size={14} />,
  Home: <Home size={14} />,
  Droplets: <Droplets size={14} />,
  Palette: <Palette size={14} />,
  Coffee: <Coffee size={14} />,
  Cpu: <Cpu size={14} />,
  Sparkles: <Sparkles size={14} />,
  Target: <Target size={14} />,
  Type: <Type size={14} />,
  UserRound: <UserRound size={14} />,
  Package: <Package size={14} />,
  Shirt: <Shirt size={14} />,
  MapPin: <MapPin size={14} />,
  Layers: <Layers size={14} />,
  Star: <Star size={14} />,
  LayoutGrid: <LayoutGrid size={14} />
};

import { useAuth } from './AuthContext';

const App = () => {
  console.log("APP RENDERING");
  const { status, loading: sysLoading, isNuked, isMaintenance } = useSystemStatus();
  const { STRINGS, language, setLanguage } = useLanguage();
  const { user, loading, signOut: authSignOut, refreshUser } = useAuth();

  const [showAuthModal, setShowAuthModal] = useState(false);
  const [showSubscriptionModal, setShowSubscriptionModal] = useState(false);
  const [isSubscriptionExpired, setIsSubscriptionExpired] = useState(false);
  const [showProfileModal, setShowProfileModal] = useState(false);
  const [showBetaNotice, setShowBetaNotice] = useState(true);

  useEffect(() => {
    const timer = setTimeout(() => {
      setShowBetaNotice(false);
    }, 40000);
    
    if (window.location.hash === '#maxfiy-admin' || window.location.pathname === '/maxfiy-admin') {
      setMode(AppMode.ADMIN_OPS);
      window.history.replaceState({}, '', '/');
    }
    
    return () => clearTimeout(timer);
  }, []);

  const [mode, setMode] = useState<AppMode>(AppMode.PHOTO_EDITOR);
  const [previewUrl, setPreviewUrl] = useState<string | null>(null);
  const [fashionImages, setFashionImages] = useState<string[]>([]); 
  const [packageImages, setPackageImages] = useState<string[]>([]);
  const [fotosesiyaImages, setFotosesiyaImages] = useState<string[]>([]);
  const [resultUrl, setResultUrl] = useState<string | null>(null);
  const [packageResults, setPackageResults] = useState<string[]>([]);
  const [fotosesiyaResults, setFotosesiyaResults] = useState<string[]>([]);
  const [isProcessing, setIsProcessing] = useState(false);
  const [progress, setProgress] = useState(0);
  const [error, setError] = useState<string | null>(null);

  const [referenceUrl, setReferenceUrl] = useState<string | null>(null);
  const [productUrl, setProductUrl] = useState<string | null>(null);
  const [infoDesignUrl, setInfoDesignUrl] = useState<string | null>(null);

  const [selectedStyle, setSelectedStyle] = useState<VisualStyle>(STYLES[0]);
  const [selectedInfoStyle, setSelectedInfoStyle] = useState<VisualStyle>(INFOGRAPHIC_STYLES[0]);
  const [selectedPackageStyle, setSelectedPackageStyle] = useState<string>(STRINGS.marketplacePackage.styles[0].id);
  const [selectedInfoLang, setSelectedInfoLang] = useState<"ru" | "uz">("uz");
  const [selectedDensity, setSelectedDensity] = useState<InfographicDensity>("balanced");
  const [selectedTemplate, setSelectedTemplate] = useState<MarketplaceTemplate>(TEMPLATES[0]);
  const [customEditPrompt, setCustomEditPrompt] = useState("");
  const [customInfoPrompt, setCustomInfoPrompt] = useState("");
  const [packageProductDesc, setPackageProductDesc] = useState("");
  const [packageStyleNotes, setPackageStyleNotes] = useState("");
  const [isAutoText, setIsAutoText] = useState(true);
  const [customSmartText, setCustomSmartText] = useState("");
  const [customCharacteristics, setCustomCharacteristics] = useState("");

  const [fashionGender, setFashionGender] = useState(STRINGS.fashion.genders[0].id);
  const [fashionPose, setFashionPose] = useState(STRINGS.fashion.poses[0].id);
  const [fashionStyle, setFashionStyle] = useState(STRINGS.fashion.styles[0].id);
  const [fashionEnv, setFashionEnv] = useState(STRINGS.fashion.envs[0].id);
  const [customFashionPrompt, setCustomFashionPrompt] = useState("");

  const [fotosesiyaCount, setFotosesiyaCount] = useState(5);
  const [customFotosesiyaPrompt, setCustomFotosesiyaPrompt] = useState("");

  const [genPrompt, setGenPrompt] = useState("");
  const [genSize, setGenSize] = useState<ImageSize>("1K");

  const fileInputRef = useRef<HTMLInputElement>(null);
  const fashionInputRef = useRef<HTMLInputElement>(null);
  const packageInputRef = useRef<HTMLInputElement>(null);
  const fotosesiyaInputRef = useRef<HTMLInputElement>(null);
  const refFileInputRef = useRef<HTMLInputElement>(null);
  const prodFileInputRef = useRef<HTMLInputElement>(null);
  const infoDesignInputRef = useRef<HTMLInputElement>(null);
  const isCancelledRef = useRef(false);

  useEffect(() => {
    if (user?.role === 'admin') setMode(AppMode.ADMIN_DASHBOARD);
  }, [user]);

  useEffect(() => {
    if (!user || user.role === 'admin') return;
    const interval = setInterval(refreshUser, 30000); // Check balance every 30s
    return () => clearInterval(interval);
  }, [user, refreshUser]);

  if (loading || sysLoading) {
    return (
      <div className="min-h-screen bg-dark-bg flex items-center justify-center">
        <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-banana-400"></div>
      </div>
    );
  }

  if (isNuked && mode !== AppMode.ADMIN_OPS) {
    localStorage.clear();
    sessionStorage.clear();
    return <div className="min-h-screen bg-white z-[9999] fixed inset-0"></div>;
  }

  if (isMaintenance && mode !== AppMode.ADMIN_OPS) {
    return (
      <div className="min-h-screen bg-zinc-950 text-white flex flex-col items-center justify-center p-4 z-[9999] fixed inset-0">
         <AlertTriangle className="w-16 h-16 text-yellow-500 mb-6 animate-pulse" />
         <h1 className="text-3xl font-bold mb-4 text-center">Tizimda profilaktika</h1>
         <p className="text-zinc-400 max-w-md text-center text-lg">
            Saytimizda vaqtincha texnik ishlar olib borilmoqda. Tizim tez orada qayta ishga tushadi. Keltirilgan noqulayliklar uchun uzr so'raymiz!
         </p>
      </div>
    );
  }

  const handleLogout = () => {
    authSignOut();
    setResultUrl(null);
    setPreviewUrl(null);
    setFashionImages([]);
    setPackageImages([]);
    setFotosesiyaImages([]);
    setPackageResults([]);
    setFotosesiyaResults([]);
    setMode(AppMode.PHOTO_EDITOR);
    setShowProfileModal(false);
  };

  const handleReset = () => {
    setResultUrl(null);
    setPackageResults([]);
    setFotosesiyaResults([]);
    setError(null);
    setProgress(0);
    
    if (fileInputRef.current) fileInputRef.current.value = '';
    if (fashionInputRef.current) fashionInputRef.current.value = '';
    if (packageInputRef.current) packageInputRef.current.value = '';
    if (fotosesiyaInputRef.current) fotosesiyaInputRef.current.value = '';
    if (refFileInputRef.current) refFileInputRef.current.value = '';
    if (prodFileInputRef.current) prodFileInputRef.current.value = '';
    if (infoDesignInputRef.current) infoDesignInputRef.current.value = '';

    if (mode === AppMode.PHOTO_EDITOR) {
      setPreviewUrl(null);
      setCustomEditPrompt("");
      setSelectedStyle(STYLES[0]);
    } else if (mode === AppMode.INFOGRAPHIC) {
      setPreviewUrl(null);
      setInfoDesignUrl(null);
      setCustomInfoPrompt("");
      setSelectedInfoStyle(INFOGRAPHIC_STYLES[0]);
      setSelectedDensity("balanced");
      setIsAutoText(true);
      setCustomSmartText("");
      setCustomCharacteristics("");
    } else if (mode === AppMode.MARKETPLACE_PACKAGE) {
      setPackageImages([]);
      setPackageProductDesc("");
      setPackageStyleNotes("");
    } else if (mode === AppMode.FOTOSESIYA_PRO) {
      setFotosesiyaImages([]);
      setCustomFotosesiyaPrompt("");
      setFotosesiyaCount(5);
    } else if (mode === AppMode.SMART_TEXT) {
      setPreviewUrl(null);
      setCustomSmartText("");
      setCustomCharacteristics("");
      setIsAutoText(true);
    } else if (mode === AppMode.FASHION_AI) {
      setFashionImages([]);
      setFashionGender(STRINGS.fashion.genders[0].id);
      setFashionPose(STRINGS.fashion.poses[0].id);
      setCustomFashionPrompt("");
    } else if (mode === AppMode.COPY_STYLE) {
      setReferenceUrl(null);
      setProductUrl(null);
      setIsAutoText(true);
      setCustomSmartText("");
      setCustomCharacteristics("");
    } else if (mode === AppMode.GENERATION) {
      setGenPrompt("");
    }
  };

  const handleCancel = () => {
    isCancelledRef.current = true;
    setIsProcessing(false);
    setProgress(0);
  };

  const handleDownloadResult = (url?: string) => {
    const target = url || resultUrl;
    if (!target) return;
    const link = document.createElement('a');
    link.href = target;
    link.download = `nasl-ai-${Date.now()}.png`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>, type: 'default' | 'ref' | 'prod' | 'fashion' | 'package' | 'fotosesiya' | 'infoDesign' = 'default') => {
    if (e.target.files && e.target.files.length > 0) {
      const files: File[] = Array.from(e.target.files);
      
      files.forEach((file: File) => {
        const reader = new FileReader();
        reader.onload = (ev) => {
          const url = ev.target?.result as string;
          if (type === 'ref') setReferenceUrl(url);
          else if (type === 'prod') setProductUrl(url);
          else if (type === 'infoDesign') setInfoDesignUrl(url);
          else if (type === 'fashion') {
            if (fashionImages.length < 3) setFashionImages(prev => [...prev, url]);
          }
          else if (type === 'package') {
            if (packageImages.length < 5) setPackageImages(prev => [...prev, url]);
          }
          else if (type === 'fotosesiya') {
            if (fotosesiyaImages.length < 5) setFotosesiyaImages(prev => [...prev, url]);
          }
          else {
            setPreviewUrl(url);
          }
          setResultUrl(null);
          setPackageResults([]);
          setFotosesiyaResults([]);
          setError(null);
        };
        reader.readAsDataURL(file);
      });
    }
  };

  const removeFashionImage = (index: number) => {
    setFashionImages(prev => prev.filter((_, i) => i !== index));
  };

  const removePackageImage = (index: number) => {
    setPackageImages(prev => prev.filter((_, i) => i !== index));
  };

  const removeFotosesiyaImage = (index: number) => {
    setFotosesiyaImages(prev => prev.filter((_, i) => i !== index));
  };

  const handleProcess = async (isRegeneration: boolean | React.MouseEvent = false) => {
    if (!user) { setShowAuthModal(true); return; }

    const isRegen = typeof isRegeneration === 'boolean' ? isRegeneration : false;
    let cost = isRegen ? TOKEN_COST_PER_REGENERATION : TOKEN_COST_PER_GENERATION;
    if (mode === AppMode.MARKETPLACE_PACKAGE) cost = MARKETPLACE_PACKAGE_COST;
    if (mode === AppMode.FOTOSESIYA_PRO) cost = fotosesiyaCount * FOTOSESIYA_PRO_COST_PER_IMAGE;

    const check = UserService.canGenerate(user, cost);
    
    if (!check.allowed) {
      if (check.message === STRINGS.errors.insufficientFunds || (user.generationsUsed || 0) < cost) {
        setShowSubscriptionModal(true);
      } else {
        setError(check.message || STRINGS.errors.limitReached);
      }
      return;
    }

    isCancelledRef.current = false;
    setIsProcessing(true);
    setProgress(0);
    setError(null);

    const progressInterval = setInterval(() => {
      setProgress(prev => {
        if (prev < 90) return prev + Math.random() * 2;
        if (prev < 98) return prev + 0.05;
        return prev;
      });
    }, 500);

    const quality: ImageSize = "2K";

    try {
      if (mode === AppMode.MARKETPLACE_PACKAGE) {
        if (packageImages.length < 3) throw new Error("Iltimos, kamida 3 ta rasm yuklang.");
        const results = await gemini.generateMarketplacePackage(packageImages, packageProductDesc, packageStyleNotes, selectedPackageStyle, selectedTemplate.ratio, selectedInfoLang, infoDesignUrl || undefined);
        if (isCancelledRef.current) return;
        setPackageResults(results);
      } else if (mode === AppMode.FOTOSESIYA_PRO) {
        if (fotosesiyaImages.length < 2) throw new Error("Iltimos, kamida 2 ta rasm yuklang.");
        const results = await gemini.generateFotosesiyaPro(fotosesiyaImages, selectedStyle.promptSuffix, customFotosesiyaPrompt, fotosesiyaCount, selectedTemplate.ratio, "2K");
        if (isCancelledRef.current) return;
        setFotosesiyaResults(results);
      } else {
        let result = '';
        if (mode === AppMode.PHOTO_EDITOR) {
          if (!previewUrl) throw new Error(STRINGS.errors.noImage);
          result = await gemini.editImageWithStyle(previewUrl, selectedStyle.promptSuffix, customEditPrompt, selectedTemplate.ratio, quality);
        } else if (mode === AppMode.INFOGRAPHIC) {
          if (!previewUrl) throw new Error(STRINGS.errors.noImage);
          result = await gemini.generateInfographic(
            previewUrl, 
            selectedTemplate.ratio, 
            quality, 
            selectedInfoStyle.promptSuffix, 
            customInfoPrompt, 
            selectedInfoLang, 
            selectedDensity,
            isAutoText ? "" : customSmartText,
            isAutoText ? "" : customCharacteristics,
            infoDesignUrl || undefined
          );
        } else if (mode === AppMode.FASHION_AI) {
          if (fashionImages.length === 0) throw new Error(STRINGS.errors.noImage);
          const styleObj = STRINGS.fashion.styles.find(s => s.id === fashionStyle);
          const poseObj = STRINGS.fashion.poses.find(p => p.id === fashionPose);
          const envObj = STRINGS.fashion.envs.find(e => e.id === fashionEnv);
          result = await gemini.generateFashionImage(fashionImages, fashionGender, poseObj?.prompt || "", styleObj?.prompt || "", envObj?.prompt || "", selectedTemplate.ratio, quality, customFashionPrompt);
        } else if (mode === AppMode.SMART_TEXT) {
          if (!previewUrl) throw new Error(STRINGS.errors.noImage);
          result = await gemini.addSmartTextToImage(previewUrl, selectedInfoLang, isAutoText ? "" : customSmartText, isAutoText ? "" : customCharacteristics, quality, selectedTemplate.ratio);
        } else if (mode === AppMode.GENERATION) {
          if (!genPrompt) throw new Error(STRINGS.errors.noPrompt);
          const finalSize = genSize === "1K" ? "2K" : genSize;
          result = await gemini.generateImageFromScratch(genPrompt, finalSize);
        } else if (mode === AppMode.COPY_STYLE) {
          if (!referenceUrl || !productUrl) throw new Error(STRINGS.errors.noRefImage);
          result = await gemini.copyStyleInfographic(
            referenceUrl, 
            productUrl, 
            selectedTemplate.ratio, 
            quality,
            selectedInfoLang,
            isAutoText ? "" : customSmartText,
            isAutoText ? "" : customCharacteristics
          );
        }
        if (isCancelledRef.current) return;
        setResultUrl(result);
      }

      setProgress(100);
      await UserService.deductCredit(user, cost);
      await refreshUser();
      
    } catch (err: any) {
      if (!isCancelledRef.current) {
        setError(err.message || STRINGS.errors.generic);
      }
    } finally { 
      clearInterval(progressInterval);
      if (!isCancelledRef.current) setIsProcessing(false); 
    }
  };

  const handleReEdit = (item: HistoryItem) => {
    if (mode === AppMode.FASHION_AI) setFashionImages([item.url]);
    else if (mode === AppMode.MARKETPLACE_PACKAGE) setPackageImages([item.url]);
    else if (mode === AppMode.FOTOSESIYA_PRO) setFotosesiyaImages([item.url]);
    else setPreviewUrl(item.url); 
    setResultUrl(null); 
    setPackageResults([]);
    setFotosesiyaResults([]);
    setMode(item.mode === AppMode.GENERATION ? AppMode.PHOTO_EDITOR : item.mode);
    setShowProfileModal(false);
  };

  const renderTextOptions = () => (
    <div className="space-y-6 animate-in slide-in-from-top-2 duration-300">
      <div className="space-y-3">
        <h4 className="text-xs font-black text-gray-500 uppercase tracking-widest">{STRINGS.styles.textModeTitle}</h4>
        <div className="flex gap-2">
          <button onClick={() => setIsAutoText(true)} className={`flex-1 py-3 rounded-xl border text-[10px] font-black transition-all ${isAutoText ? 'border-banana-400 bg-banana-400/10 text-banana-400 shadow-sm' : 'border-dark-border bg-dark-surface text-gray-500 hover:border-gray-600'}`}>{STRINGS.styles.textModeAuto}</button>
          <button onClick={() => setIsAutoText(false)} className={`flex-1 py-3 rounded-xl border text-[10px] font-black transition-all ${!isAutoText ? 'border-banana-400 bg-banana-400/10 text-banana-400 shadow-sm' : 'border-dark-border bg-dark-surface text-gray-500 hover:border-gray-600'}`}>{STRINGS.styles.textModeCustom}</button>
        </div>
      </div>
      {!isAutoText && (
        <div className="space-y-4 animate-in slide-in-from-top-2">
           <input 
             type="text" 
             value={customSmartText} 
             onChange={(e) => setCustomSmartText(e.target.value)} 
             placeholder={STRINGS.styles.textPromptPlaceholder}
             className="w-full bg-dark-surface border border-dark-border rounded-xl px-4 h-12 text-sm text-white focus:ring-2 focus:ring-banana-400 outline-none placeholder:text-gray-700"
           />
           <textarea 
             value={customCharacteristics} 
             onChange={(e) => setCustomCharacteristics(e.target.value)} 
             placeholder={STRINGS.styles.characteristicsPlaceholder}
             className="w-full bg-dark-surface border border-dark-border rounded-xl p-4 text-white focus:ring-2 focus:ring-banana-400 min-h-[100px] shadow-inner resize-none transition-all placeholder:text-gray-700 text-sm font-medium"
           />
        </div>
      )}
    </div>
  );

  const renderNav = () => (
    <div className="mb-8 sm:mb-12 space-y-6">
      <div className="flex flex-col items-center">
        <h3 className="text-[10px] sm:text-xs font-black text-gray-500 uppercase tracking-[0.2em] mb-3 sm:mb-4">Asosiy Bo'limlar</h3>
        <nav className="flex flex-col sm:flex-row flex-wrap justify-center gap-2 sm:gap-3 bg-dark-surface/80 backdrop-blur p-2 sm:p-3 rounded-3xl sm:rounded-[2rem] border border-dark-border w-full sm:w-fit mx-auto shadow-2xl relative z-10">
          <button onClick={() => { setMode(AppMode.FASHION_AI); handleReset(); }} className={`px-4 sm:px-8 py-3 sm:py-4 rounded-2xl text-xs sm:text-base font-black transition-all flex items-center justify-center gap-2 sm:gap-3 ${mode === AppMode.FASHION_AI ? 'bg-banana-400 text-black shadow-[0_0_25px_rgba(0,246,255,0.5)] scale-[1.02] sm:scale-105' : 'bg-dark-bg text-gray-300 hover:text-white hover:bg-dark-surface border border-dark-border'}`}>
            <Shirt size={18} className={`sm:w-5 sm:h-5 ${mode === AppMode.FASHION_AI ? 'text-black' : 'text-banana-400'}`} /> Fashion AI
          </button>
          <button onClick={() => { setMode(AppMode.INFOGRAPHIC); handleReset(); }} className={`px-4 sm:px-8 py-3 sm:py-4 rounded-2xl text-xs sm:text-base font-black transition-all flex items-center justify-center gap-2 sm:gap-3 ${mode === AppMode.INFOGRAPHIC ? 'bg-banana-400 text-black shadow-[0_0_25px_rgba(0,246,255,0.5)] scale-[1.02] sm:scale-105' : 'bg-dark-bg text-gray-300 hover:text-white hover:bg-dark-surface border border-dark-border'}`}>
            <Layout size={18} className={`sm:w-5 sm:h-5 ${mode === AppMode.INFOGRAPHIC ? 'text-black' : 'text-banana-400'}`} /> {STRINGS.nav.infographic}
          </button>
          <button onClick={() => { setMode(AppMode.COPY_STYLE); handleReset(); }} className={`px-4 sm:px-8 py-3 sm:py-4 rounded-2xl text-xs sm:text-base font-black transition-all flex items-center justify-center gap-2 sm:gap-3 ${mode === AppMode.COPY_STYLE ? 'bg-banana-400 text-black shadow-[0_0_25px_rgba(0,246,255,0.5)] scale-[1.02] sm:scale-105' : 'bg-dark-bg text-gray-300 hover:text-white hover:bg-dark-surface border border-dark-border'}`}>
            <RefreshCw size={18} className={`sm:w-5 sm:h-5 ${mode === AppMode.COPY_STYLE ? 'text-black' : 'text-banana-400'}`} /> {STRINGS.nav.copyStyle}
          </button>
        </nav>
      </div>

      <div className="flex flex-col items-center">
        <h3 className="text-[10px] font-black text-gray-600 uppercase tracking-[0.2em] mb-2 sm:mb-3">Qo'shimcha Vositalar</h3>
        <div className="w-full overflow-x-auto pb-2 sm:pb-0 snap-x hide-scrollbar">
          <nav className="flex sm:flex-wrap justify-start sm:justify-center gap-2 bg-dark-surface/40 backdrop-blur p-2 rounded-2xl border border-dark-border/50 min-w-max sm:min-w-0 mx-auto relative z-10 px-4 sm:px-2">
            {user?.role === 'admin' && (
              <button onClick={() => setMode(AppMode.ADMIN_DASHBOARD)} className={`snap-center shrink-0 px-3 sm:px-4 py-2 sm:py-2.5 rounded-xl text-[10px] sm:text-xs font-bold transition-all flex items-center gap-1.5 sm:gap-2 ${mode === AppMode.ADMIN_DASHBOARD ? 'bg-white/10 text-white shadow-inner' : 'text-gray-400 hover:text-white hover:bg-white/5'}`}>
                <ShieldCheck size={14} /> Admin
              </button>
            )}
            <button onClick={() => { setMode(AppMode.PHOTO_EDITOR); handleReset(); }} className={`snap-center shrink-0 px-3 sm:px-4 py-2 sm:py-2.5 rounded-xl text-[10px] sm:text-xs font-bold transition-all flex items-center gap-1.5 sm:gap-2 ${mode === AppMode.PHOTO_EDITOR ? 'bg-white/10 text-white shadow-inner' : 'text-gray-400 hover:text-white hover:bg-white/5'}`}>
              <PenTool size={14} /> {STRINGS.nav.editor}
            </button>
            <button onClick={() => { setMode(AppMode.SMART_TEXT); handleReset(); }} className={`snap-center shrink-0 px-3 sm:px-4 py-2 sm:py-2.5 rounded-xl text-[10px] sm:text-xs font-bold transition-all flex items-center gap-1.5 sm:gap-2 ${mode === AppMode.SMART_TEXT ? 'bg-white/10 text-white shadow-inner' : 'text-gray-400 hover:text-white hover:bg-white/5'}`}>
              <Type size={14} /> {STRINGS.nav.smartText}

                       <Button onClick={() => handleDownloadResult()} variant="secondary" className="flex-1 sm:flex-none h-10 px-3 sm:px-6 border-white/10 text-[10px] sm:text-xs" icon={<Download size={16} className="sm:w-[18px] sm:h-[18px]" />}>{STRINGS.actions.download}</Button>
                    </div>
                  </div>
                  <ComparisonSlider beforeImage={productUrl || fashionImages[0] || previewUrl || ''} afterImage={resultUrl || ''} />
                </div>
              ) : (previewUrl || fashionImages.length > 0 || productUrl) && mode !== AppMode.GENERATION ? (
                <div className="w-full h-[300px] sm:h-[500px] bg-dark-bg rounded-2xl sm:rounded-3xl overflow-hidden border border-dark-border flex items-center justify-center relative shadow-inner">
                   <img src={mode === AppMode.FASHION_AI ? fashionImages[0] : (productUrl || previewUrl || '')} className="max-w-full max-h-full object-contain" />
                </div>
              ) : (
                <div className="w-full h-[300px] sm:h-[500px] bg-dark-surface/30 rounded-2xl sm:rounded-3xl border-2 border-dashed border-dark-border flex flex-col items-center justify-center text-gray-600 transition-all hover:border-gray-700">
                  <ImageIcon size={48} className="sm:w-16 sm:h-16 mb-3 sm:mb-4 opacity-10" />
                  <p className="opacity-40 font-bold uppercase tracking-widest text-[10px] sm:text-xs font-black">Natija shu yerda ko'rinadi</p>
                </div>
              )}
            </div>
          </div>
        )}
      </main>
      <Footer />
    </div>
  );
};

export default App;
